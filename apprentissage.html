    if ((sock = socket(AF_INET, SOCK_STREAM, 0)) == -1) {
        perror("socket");

    }

    local.sin_family = AF_INET;
    local.sin_port = htons(PORT);
    local.sin_addr.s_addr = INADDR_ANY;

        perror("bind");
        exit(EXIT_FAILURE);
    }

    if (listen(sock, 5) == -1) {
        perror("listen");
        exit(EXIT_FAILURE);
    }

    printf("Serveur en écoute sur le port %d...\n", PORT);

    while (1) {
        lg = sizeof(distant);
        if ((client_sock = accept(sock, (struct sockaddr *)&distant, &lg)) == -1) {
            perror("accept");
            exit(EXIT_FAILURE);
        }
        printf("Client connecté.\n");
        if (fork() == 0) {
            close(sock);

                memset(mess, 0, BUFFER_SIZE);

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <errno.h>
#include <netinet/in.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <arpa/inet.h>

#define PORT 12345
#define BUFFER_SIZE 80

int main() {
    int sock;
    struct sockaddr_in serv_addr;
    char mess[BUFFER_SIZE];

    if ((sock = socket(AF_INET, SOCK_STREAM, 0)) == -1) {
        perror("socket");
        exit(EXIT_FAILURE);
    }

    memset(&serv_addr, 0, sizeof(serv_addr));
    serv_addr.sin_family = AF_INET;
    serv_addr.sin_port = htons(PORT);
    serv_addr.sin_addr.s_addr = inet_addr("127.0.0.1");

    if (connect(sock, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) == -1) {
        perror("connect");
        exit(EXIT_FAILURE);
    }

    printf("Connecté au serveur. Tapez 'fin' pour quitter.\n");

    while (1) {
        printf("Vous : ");
        fgets(mess, BUFFER_SIZE, stdin);
        mess[strcspn(mess, "\n")] = '\0';

        if (write(sock, mess, strlen(mess)) == -1) {
            perror("write");
            break;
        }

        if (strncmp(mess, "fin", 3) == 0) break;
    }

    close(sock);
    return 0;
}
